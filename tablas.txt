-- tablas de una base de datos Milco
CREATE TABLE proveedor(
id_proveedor	INT(11) AUTO_INCREMENT NOT NULL,
nombre		VARCHAR(50) NOT NULL,
correo		VARCHAR(25) NULL,
telefono	INT(20) NOT NULL,
direccion	VARCHAR(50) NOT NULL,
representanten	VARCHAR(50) NOT NULL,
telefono_rep	INT(9)NOT NULL,
detalle		VARCHAR(80) NULL,
PRIMARY KEY(id_proveedor)
)ENGINE = INNODB; CHARSET=utf8

CREATE TABLE producto(
id_producto	INT(11) AUTO_INCREMENT NOT NULL,
codigo_prod	VARCHAR(11) NOT NULL,
nombre_prod	VARCHAR(30) NOT NULL,
cantidad_prod_compra	INT(5) NOT NULL,
marca_prod	VARCHAR(30) NULL,
precio_venta_uni	DECIMAL(11.2) NOT NULL,
fecha_ingreso	TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY(id_producto)
)ENGINE = INNODB;


CREATE TABLE compra(
id_compra	INT(11) AUTO_INCREMENT NOT NULL,
precio_compra	DECIMAL(11.2) NOT NULL,
impuesto	DECIMAL(11.2) NOT NULL,
total		DECIMAL(11.2) NOT NULL,
fecha		TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
detalle		VARCHAR(80)NULL,
PRIMARY KEY(id_compra)
)ENGINE = INNODB;

CREATE TABLE clase(
id_clase	INT(11) AUTO_INCREMENT NOT NULL,
nombre_cla	VARCHAR(30) NOT NULL,
detalle		VARCHAR(30) NULL,
CONSTRAINT PK_clase PRIMARY KEY (id_clase)
)ENGINE = INNODB;

CREATE TABLE categoria(
id_categoria	INT(11) AUTO_INCREMENT NOT NULL,
nombre_cate	VARCHAR(30) NOT NULL,
detalle		VARCHAR(30) NULL,
CONSTRAINT PK_categoria PRIMARY KEY(id_categoria)
)ENGINE = INNODB;

CREATE TABLE trabajador(
id_trabajador	INT(11)AUTO_INCREMENT NOT NULL,
nombre		VARCHAR(10) NOT NULL,
apellido_pa	VARCHAR(10) NOT NULL,
apellido_ma	VARCHAR(10) NOT NULL,
dni		INT(8) NOT NULL,
telefono	INT(9) NOT NULL,
direccion	VARCHAR(50) NOT NULL,
correo		VARCHAR(25) NULL,
estado_civil	VARCHAR(10) NULL,
codigo_trab	INT(5)NOT NULL, 
fecha_ingreso	TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
estado		VARCHAR(10) NOT NULL,
PRIMARY KEY(id_trabajador)
)ENGINE = INNODB;

CREATE TABLE vendedor(
id_vendedor	INT(11)AUTO_INCREMENT NOT NULL,
usuario		VARCHAR(20) NOT NULL,
passwor		VARCHAR(10) NOT NULL,
detalle		VARCHAR(50) NULL,
PRIMARY KEY(id_vendedor)
)ENGINE = INNODB;

CREATE TABLE cliente(
id_cliente	INT(11) AUTO_INCREMENT NOT NULL,
dni		INT(8) NOT NULL,
nombre		VARCHAR(30) NULL,
detalle		VARCHAR(80) NULL,
PRIMARY KEY(id_cliente)
)ENGINE = INNODB;

CREATE TABLE venta(
id_venta	INT(11) AUTO_INCREMENT NOT NULL,		
cantidad	DECIMAL(11.2) NOT NULL,
impuesto	DECIMAL(11.2) NOT NULL,
total		DECIMAL(11.2) NOT NULL,
PRIMARY KEY(id_venta)
)ENGINE = INNODB;

CREATE TABLE Retiro(
id_retiro	INT(11) AUTO_INCREMENT,
fecha_salida	TIMESTAMP DEFAULT CURRENT_TIMESTAMP,	
detalle_reti	VARCHAR(100) NOT NULL,
PRIMARY KEY(id_retiro)
)ENGINE = INNODB;
